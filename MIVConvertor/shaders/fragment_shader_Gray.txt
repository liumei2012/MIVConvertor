#version 300 es
precision highp float;
precision highp int;
out vec4 outColor; // name of the output variable

in vec2 interpolatedTexCoord;

uniform sampler2D MyTex; // min_filter="NEAREST" mag_filter="NEAREST"

uniform int width;
uniform int height;
uniform int samples; // description="Number of samples" defaultval="256"
uniform bool debug; // description="Enable debugging of sampling locations" defaultval="false"

uniform int inputImageWidth;
uniform int inputImageHeight;


#define PI 3.1415926535897932384626433832795



void main() {
  // Env samp
  vec3 texColor = texture(MyTex, interpolatedTexCoord).rgb;
  float gray = dot(vec3(0.2126, 0.7152, 0.0722), texColor.rgb);
  outColor.rgb = vec3(gray, gray, gray);
  outColor.a = 1.0;
}
