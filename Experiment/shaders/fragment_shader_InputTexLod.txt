#version 300 es
precision highp float;
out vec4 outColor; 
// name of the output variable

in vec2 interpolatedTexCoord;
uniform sampler2D MyTex; 
// min_filter="NEAREST" mag_filter="NEAREST"

void main() {
  // Env samp

  //vec3 texColor = texture(myTexture, interpolatedTexCoord).rgb;
  vec3 texColor = textureLod(MyTex, interpolatedTexCoord, 5.0).rgb;
  outColor.rgb = vec3(texColor.r, texColor.g, texColor.b);
  outColor.a = 1.0;
}
