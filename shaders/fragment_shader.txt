#version 300 es
precision highp float;
out vec4 outColor; // name of the output variable

in vec2 interpolatedTexCoord;

uniform sampler2D myTexture; // mag_filter="NEAREST" min_filter="LINEAR_MIPMAP_LINEAR"
uniform float level;

void main() {
  vec4 texColor = textureLod(myTexture, interpolatedTexCoord, 5.0);
  outColor = texColor;
}
